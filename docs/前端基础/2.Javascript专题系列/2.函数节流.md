```js
const throttle = function(func, wait) {
  let timer = null;

  return function() {
    let context = this;
    let args = arguments;

    if (!timer) {
      timer = setTimeout(() => {
        func.apply(context, args);
        clearTimeout(timer);
        timer = null;
      }, wait);
    }
  }
}
```
